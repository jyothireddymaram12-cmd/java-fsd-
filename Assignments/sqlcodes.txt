4. CHECK
Syntax
CREATE TABLE Student (
    id INT,
    age INT CHECK(age>=18)
);
Insert
CREATE TABLE Student (
    id INT,
    age INT CHECK(age>=18)
);
INSERT INTO Student VALUES(1,20);
Error Example
CREATE TABLE Student (
    id INT,
    age INT CHECK(age>=18)
);
INSERT INTO Student VALUES(1,20);
INSERT INTO Student VALUES(2,15);
Error
Check constraint violated
Solution
CREATE TABLE Student (
    id INT,
    age INT CHECK(age>=18)
);
INSERT INTO Student VALUES(1,20);
INSERT INTO Student VALUES(2,18);
